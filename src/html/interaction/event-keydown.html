キーボードイベント


    window.addEventListener("keydown", function(event) {
        // キーコード(どのキーが押されたか)を取得
        var keyCode = event.keyCode;
        if (keyCode === 39) {
            console.log("押されたキー : 右 (" + keyCode + ")");
        } else if (keyCode === 37) {
            console.log("押されたキー : 左 (" + keyCode + ")");
        } else if (keyCode === 38) {
            console.log("押されたキー : 上 (" + keyCode + ")");
        } else if (keyCode === 40) {
            console.log("押されたキー : 下 (" + keyCode + ")");
        }
    });
    window.addEventListener("keyup", function(event) {
        console.log("キーボードが離された");
    });
