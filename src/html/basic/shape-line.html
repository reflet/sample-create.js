線と塗り


    // シェイプ作成 (塗りつぶし)
    shape.graphics.beginFill("DarkRed");   // 色を塗る
    shape.graphics.drawCircle(0, 0, 80);   // 円を描く (中心のX座標, 中心のY座標, 半径)

    // シェイプ作成 (枠線)
    shape.graphics.beginStroke("DarkRed"); // 塗りの色を指定する
    shape.graphics.setStrokeStyle(5);      // 線の幅を指定する
    shape.graphics.drawCircle(0, 0, 80);   // 円を描く (中心のX座標, 中心のY座標, 半径)
