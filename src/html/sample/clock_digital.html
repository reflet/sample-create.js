サンプル(デジタル時計)


    // ラベルを追加する
    var label = new createjs.Text("", "80px sans-serif", "red");
    stage.addChild(label);

    // 時間経過のイベント
    createjs.Ticker.addEventListener("tick", function() {
        // 時間を2桁の数値で生成する
        var now = new Date(),
            h = ('00' + now.getHours()).slice(-2),   // 時(00〜23)
            m = ('00' + now.getMinutes()).slice(-2), // 分(00〜59)
            s = ('00' + now.getSeconds()).slice(-2); // 秒(00〜59)

        // ラベルの文字列を更新する
        label.text = h + ":" + m + ":" + s;

        // 描画更新
        stage.update();
    });
