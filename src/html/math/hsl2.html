HSLカラー(色相と彩度と明度を順番に配置する)


    // 色相(HSLカラー)を算出する
    hue = i * (360 / MAX); // 0〜360を等間隔に20等分した値
    color = "hsl(" + hue + ", 100%, 50%)";

    // 彩度(HSLカラー)を算出する
    saturation = i * (100 / MAX); // 0〜100%を等間隔に20等分した値
    color = "hsl(0, " + saturation + "%, 50%)";

    // 明度(HSLカラー)を算出する
    luminance = i * (100 / MAX); // 0〜100%を等間隔に20等分した値
    color = "hsl(0, 100%, " + luminance + "%)";
