HSLカラー(ランダムな色の円を配置する)


    // HSLカラーを算出する(ランダム 0〜360)
    var hue = 360 * Math.random(),
        color = "hsl(" + hue + ", 100%, 50%)";

    // 座標計算
    var x = stage.canvas.width * Math.random(),  // ランダムな座標(X)
        y = stage.canvas.height * Math.random(); // ランダムな座標(Y)

    // 円を追加する
    var shape = new createjs.Shape();
    shape.graphics
        .clear()
        .beginFill(color)
        .drawCircle(x, y, 20);
    stage.addChild(shape);
