連続のモーションを設定する


        createjs.Tween.get({対象})
            .to({x: 940, y: 500, alpha: 0.1}, 2000) // 画面右下へ移動
            .to({x: 500, y: 0, alpha: 1.0}, 3000)   // 画面中央上へ移動
            .to({x: 500, y: 300}, 3000)             // 画面中央へ移動
            .to({scaleX: 3, scaleY: 3}, 5000);      // 300%の大きさに変形
